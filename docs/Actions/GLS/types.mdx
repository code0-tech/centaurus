# GLS Action Types

The GLS Action registers the following data types with the Hercules platform. These types are used as inputs and outputs
of the GLS functions and can be referenced in your flows.

# GLS_ADDRESS
        
<AutoTypeTable type={`
        
export type GLS_ADDRESS = {
    /**

        Primary name line (person or company).
        Max 40 characters.
         
**/
    Name1: string;
    /**

        Optional second name line (e.g., department or additional identifier).
        Max 40 characters.
         
**/
    Name2?: string;
    /**

        Optional third name line for extended address details.
        Max 40 characters.
         
**/
    Name3?: string;
    /**

        Two-letter ISO country code (e.g., DE, US).
         
**/
    CountryCode: string;
    /**

        State, province, or region.
        Optional field.
        Max 40 characters.
         
**/
    Province?: string;
    /**

        City or locality name.
        Max 40 characters.
         
**/
    City: string;
    /**

        Street name.
        Minimum 4 characters required.
         
**/
    Street: string;
    /**

        House or building number.
        Optional field.
        Max 40 characters.
         
**/
    StreetNumber?: string;
    /**

        Full name of a contact person.
        Optional field.
        Must be between 6 and 40 characters.
         
**/
    ContactPerson?: string;
    /**

        Landline phone number.
        Optional field.
        Must be between 4 and 35 characters.
         
**/
    FixedLinePhonenumber?: string;
    /**

        Mobile phone number.
        Optional field.
        Must be between 4 and 35 characters.
         
**/
    MobilePhonenumber?: string;
    /**

        Email address.
        Optional field.
        Max 80 characters.
         
**/
    eMail?: string;
    /**

        Postal or ZIP code.
        Max 10 characters.
         
**/
    ZIPCode: string;
}
        
`} name="GLS_ADDRESS"/>
